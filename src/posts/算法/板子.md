```cpp
#include <vector>
using namespace std;

long long quickPow(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp % 2 == 1) res = (res * base) % mod;
        base = (base * base) % mod;
        exp /= 2;
    }
    return res;
}

long long chineseRemainder(const vector<long long>& remainders, const vector<long long>& moduli) {
    long long prod = 1;
    for (auto m : moduli) prod *= m;

    long long result = 0;
    for (int i = 0; i < remainders.size(); i++) {
        long long pp = prod / moduli[i];
        long long inv = quickPow(pp, moduli[i] - 2, moduli[i]);
        result = (result + remainders[i] * pp % prod * inv % prod) % prod;
    }

    return result;
}
```

## 线段树

适用于区间多查多改

```cpp
#include<cstdio>
#define maxn 30005

struct Node {
    int w;
    int lazy_update;
};

Node tree[maxn];

#define LSON(x) ((x)*2)
#define RSON(x) (((x)*2)+1)
#define PARENT(x) ((x)/2)

int a[maxn];

void update_lazy(int x, int L, int R) {
    if (tree[x].lazy_update != 0) {
        tree[x].w += tree[x].lazy_update * (R-L+1);
        tree[LSON(x)].lazy_update += tree[x].lazy_update;
        tree[RSON(x)].lazy_update += tree[x].lazy_update;
        tree[x].lazy_update = 0;
    }
}

void build(int x, int L, int R) {
    if (L==R) {
        tree[x].w = a[L];
    } else {
        int mid = (L+R)/2;
        build(LSON(x), L, mid);
        build(RSON(x), mid+1, R);
        tree[x].w = tree[LSON(x)].w + tree[RSON(x)].w;
    }
}

void update(int x, int L, int R, int pos, int value) {
    if (L==R && pos == L) {
        tree[x].w = value;
    } else {
        int mid = (L+R)/2;
        if (pos <= mid) {
            update(LSON(x), L, mid, pos, value);
        } else {
            update(RSON(x), mid+1, R, pos, value);
        }
        tree[x].w = tree[LSON(x)].w + tree[RSON(x)].w;
    }
}

int query(int x, int L, int R, int QL, int QR) {
    update_lazy(x, L, R);
    if (L==QL && R==QR) {
        return tree[x].w;
    }
    int mid = (R+L)/2;
    if (QR <= mid) {
        return query(LSON(x), L, mid, QL, QR);
    } else if (QL > mid) {
        return query(RSON(x), mid+1, R, QL, QR);
    } else {
        return query(LSON(x), L, mid, QL, mid) + query(RSON(x), mid+1, R, mid+1, QR);
    }
}

void add(int x, int L, int R, int AL, int AR, int value) {
    if (L==AL && R==AR) {
        tree[x].lazy_update += value;
        return;
    }
    int mid = (L+R)/2;
    if (AR <= mid) {
        add(LSON(x), L, mid, AL, AR, value);
    } else if (mid < AL) {
        add(RSON(x), mid+1, R, AL, AR, value);
    } else {
        add(LSON(x), L, mid, AL, mid, value);
        add(RSON(x), mid+1, R, mid+1, AR, value);
    }
    tree[x].w += (AR-AL+1)*value;
}
```

## 线段树（最大子段和）
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const long long INF = 2e18;

struct Node {
    long long sum, lmax, rmax, ans;
    long long set_tag;
    int len;
};

int n, Q;
vector<Node> tree;
vector<int> arr;

void pushup(int idx) {
    int lc = idx << 1, rc = idx << 1 | 1;
    tree[idx].sum = tree[lc].sum + tree[rc].sum;
    tree[idx].lmax = max(tree[lc].lmax, tree[lc].sum + tree[rc].lmax);
    tree[idx].rmax = max(tree[rc].rmax, tree[rc].sum + tree[lc].rmax);
    tree[idx].ans = max({tree[lc].ans, tree[rc].ans, tree[lc].rmax + tree[rc].lmax});
}

void apply_set(int idx, long long v) {
    tree[idx].set_tag = v;
    tree[idx].sum = (long long)tree[idx].len * v;
    if (v >= 0) {
        tree[idx].lmax = tree[idx].rmax = tree[idx].ans = tree[idx].sum;
    } else {
        tree[idx].lmax = tree[idx].rmax = tree[idx].ans = v;
    }
}

void pushdown(int idx) {
    if (tree[idx].set_tag != INF) {
        apply_set(idx << 1, tree[idx].set_tag);
        apply_set(idx << 1 | 1, tree[idx].set_tag);
        tree[idx].set_tag = INF;
    }
}

void build(int idx, int l, int r) {
    tree[idx].set_tag = INF;
    tree[idx].len = r - l + 1;
    if (l == r) {
        long long v = arr[l];
        tree[idx].sum = v;
        if (v >= 0) {
            tree[idx].lmax = tree[idx].rmax = tree[idx].ans = v;
        } else {
            tree[idx].lmax = tree[idx].rmax = tree[idx].ans = v;
        }
        return;
    }
    int mid = l + (r - l) / 2;
    build(idx << 1, l, mid);
    build(idx << 1 | 1, mid + 1, r);
    pushup(idx);
}

void update_pos(int idx, int l, int r, int pos, long long v) {
    if (l == r) {
        tree[idx].sum = v;
        if (v >= 0) {
            tree[idx].lmax = tree[idx].rmax = tree[idx].ans = v;
        } else {
            tree[idx].lmax = tree[idx].rmax = tree[idx].ans = v;
        }
        return;
    }
    pushdown(idx);
    int mid = l + (r - l) / 2;
    if (pos <= mid) update_pos(idx << 1, l, mid, pos, v);
    else update_pos(idx << 1 | 1, mid + 1, r, pos, v);
    pushup(idx);
}

void update_range(int idx, int l, int r, int ql, int qr, long long v) {
    if (ql <= l && r <= qr) {
        apply_set(idx, v);
        return;
    }
    pushdown(idx);
    int mid = l + (r - l) / 2;
    if (ql <= mid) update_range(idx << 1, l, mid, ql, qr, v);
    if (qr > mid) update_range(idx << 1 | 1, mid + 1, r, ql, qr, v);
    pushup(idx);
}

Node query(int idx, int l, int r, int ql, int qr) {
    if (ql <= l && r <= qr) {
        return tree[idx];
    }
    pushdown(idx);
    int mid = l + (r - l) / 2;
    if (qr <= mid) return query(idx << 1, l, mid, ql, qr);
    if (ql > mid) return query(idx << 1 | 1, mid + 1, r, ql, qr);
    
    Node left = query(idx << 1, l, mid, ql, qr);
    Node right = query(idx << 1 | 1, mid + 1, r, ql, qr);
    Node res;
    res.sum = left.sum + right.sum;
    res.lmax = max(left.lmax, left.sum + right.lmax);
    res.rmax = max(right.rmax, right.sum + left.rmax);
    res.ans = max({left.ans, right.ans, left.rmax + right.lmax});
    res.len = 0; 
    return res;
}
```

## 并查集
```cpp
struct DSU {
    std::vector<int> f, siz;
    
    DSU() {}
    DSU(int n) {
        init(n);
    }
    
    void init(int n) {
        f.resize(n);
        std::iota(f.begin(), f.end(), 0);
        siz.assign(n, 1);
    }
    
    int find(int x) {
        while (x != f[x]) {
            x = f[x] = f[f[x]];
        }
        return x;
    }
    
    bool same(int x, int y) {
        return find(x) == find(y);
    }
    
    bool merge(int x, int y) {
        x = find(x);
        y = find(y);
        if (x == y) {
            return false;
        }
        siz[x] += siz[y];
        f[y] = x;
        return true;
    }
    
    int size(int x) {
        return siz[find(x)];
    }
};
```

```cpp
		dis[e[i].v]=min(dis[e[i].v],e[i].w);
	}
    while(++tot<n)
    {
        re int minn=inf;
        vis[now]=1;
        for(re int i=1;i<=n;++i)
        {
            if(!vis[i]&&minn>dis[i])
            {
                minn=dis[i];
				now=i;
            }
        }
        ans+=minn;
        for(re int i=head[now];i;i=e[i].next)
        {
        	re int v=e[i].v;
        	if(dis[v]>e[i].w&&!vis[v])
        	{
        		dis[v]=e[i].w;
        	}
		}
    }
```

```cpp
#include<bits/stdc++.h>
const int MaxN = 100010, MaxM = 500010;
struct edge
{
    int to, dis, next;
};
edge e[MaxM];

bool vis[MaxN];
int n, m, s;
inline void add_edge(int u, int v, int d)
{
    cnt++;
    e[cnt].dis = d;
    e[cnt].to = v;
    e[cnt].next = head[u];
    head[u] = cnt;
}
struct node
{
    int dis;
    int pos;
    bool operator <(const node &x) const
    {
        return x.dis < dis;
    }
};
std::priority_queue<node> q;
inline void dijkstra()
{
    dis[s] = 0;
    q.push((node){0, s});
    while(!q.empty())
    {
        node tmp = q.top();
        q.pop();
        int x = tmp.pos, d = tmp.dis;
        if(vis[x]) continue;
        vis[x] = 1;
        for(int i = head[x]; i; i = e[i].next)
        {
            int y = e[i].to;
            if(dis[y] > dis[x] + e[i].dis)
            {
                dis[y] = dis[x] + e[i].dis;
                if(!vis[y]) q.push((node){dis[y], y});
            }
        }
    }
}
int main()
{
    scanf("%d%d%d", &n, &m, &s);
    for(int i = 1; i <= n; ++i) dis[i] = 0x7fffffff;
    for(register int i = 0; i < m; ++i)
    {
        register int u, v, d;
        scanf("%d%d%d", &u, &v, &d);
        add_edge(u, v, d);
    }
    dijkstra();
    for(int i = 1; i <= n; i++) printf("%d ", dis[i]);
```

```cpp
#include <bits/stdc++.h>
```

```cpp
const ll INF = 1e18;
int n, m, s;

ll dist[100005];
bool vis[100005];

void dijkstra(int start) {
    for (int i = 1; i <= n; i++) { dist[i] = INF; vis[i] = false; }
    dist[start] = 0;
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    pq.push({0, start});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (vis[u]) continue;
        vis[u] = true;
        for (auto [v, w] : adj[u])
            if (dist[u] + w < dist[v]) { dist[v] = dist[u] + w; pq.push({dist[v], v}); }
    }
}
```
